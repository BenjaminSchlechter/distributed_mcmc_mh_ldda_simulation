61 void generate_next_cfg_vector()
62 {
63 next_cfg_vector = mc_state.cfg_vector;
64
65 bool change_happened = enforce_change;
66 do {
67 using index_t = decltype(
68 change_gen(mc_state.prn_engine,
69 next_cfg_vector).index);
70
71 // generate changes
72 std::set<index_t> changed_positions;
73
74 for (decltype(mcmc_num_changes) i = 0;
75 i < mcmc_num_changes; i++)
76 {
77 auto cfg_change =
78 change_gen(mc_state.prn_engine, next_cfg_vector);
79 next_cfg_vector.data[cfg_change.index]
80 = cfg_change.new_value;
81
82 if ((!change_happened) && (skip_unchanged_vectors))
83 {
84 changed_positions.insert(cfg_change.index);
85 }
86 }
87
88 // check if config vector is changed
89 if ((!change_happened) && (skip_unchanged_vectors)) {
90 for(auto p : changed_positions) {
91 if (next_cfg_vector.data[p]
92 != mc_state.cfg_vector.data[p])
93 {
94 change_happened = true;
95 break;
96 }
97 }
98 }
99 } while ((!change_happened) && (skip_unchanged_vectors));
100 }

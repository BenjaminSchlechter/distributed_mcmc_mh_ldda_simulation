103 value_comp_t &value_comp,
104 result_projection_t &result_projection = {})

109 state_data[new_index].prng = derive_prng(main_prng);

111 result_function_t result_fct = [&](auto... results){
112 remaining_computations--;
113 auto result = result_projection(results...);
114 static_assert(
115 std::is_same<prec_t, decltype(result)>::value
116 && "result_projection has wrong result type!");
117 state_data[new_index].value = result;
118 };

120 value_comp.prepare_computation_with_cfg_vector(
121 v, state_data[new_index].pcom);

123 auto cmd = value_comp(
124 state_data[new_index].prng,
125 scheduler.rec.rpmanager,
126 state_data[new_index].pcom);
127 cmd = scheduler.rec.on_result(cmd, result_fct);
128
129 remaining_computations++;
130 scheduler.schedule(cmd);

167 accept = true;
168 return true;
169 } else {
170 prec_t z = prob_distribution(state_data[new_index].prng);
171
172 prec_t p = acc_prob_fct(
173 state_data[old_index].value,
174 state_data[new_index].value);
175
176 bool accept = (z <= p);
177 if (accept) {
178 new_index = old_index;
179 }

182 }

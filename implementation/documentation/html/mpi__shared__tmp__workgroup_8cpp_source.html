36int mpi_shared_tmp_dir_workgroup::identify_workgroup_head()
37{
38 namespace fs = std::filesystem;
39
40 // create random tmp path
41 fs::create_directories(tmp_path);
42
43 // ensure its empty
44 assert(fs::is_empty(tmp_path));
45 assert(MPI_SUCCESS == MPI_Barrier(parent_comm));
46
47 // create dir with id inside
48 assert(fs::create_directory(
49 tmp_path + "/" + std::to_string(parent_comm_id)));
50 assert(MPI_SUCCESS == MPI_Barrier(parent_comm));
51
52 // find minimum id in local tmp dir
53 int min_id = std::numeric_limits<int>::max();
54 for (const auto & entry : fs::directory_iterator(tmp_path))
55 {
56 min_id = std::min(
57 min_id, from_string<int>(entry.path().filename()));
58 }
59
60 // sanity checks
61 assert(0 <= min_id);
62 assert(min_id < parent_comm_size);
63
64 // remove tmp directories if all are finished
65 // note: /tmp/mpi_split_by_shared_tmp/ will persist
66 MPI_Barrier(parent_comm);
67 if (parent_comm_id == min_id) {
68 std::filesystem::remove_all(tmp_path);
69 }
70
71 return min_id;
72}

80mpi_shared_tmp_dir_workgroup::mpi_shared_tmp_dir_workgroup(
81 MPI_Comm parent_comm,
82 std::string app_name
83) :
84 parent_comm(parent_comm)
85{
86 parent_comm_id = mpi_get_comm_rank(parent_comm);
87 parent_comm_size = mpi_get_comm_size(parent_comm);
88
89 if (1 == parent_comm_size)
90 { // shortcut in case of a single process
91 assert(0 == parent_comm_id);
92 parent_comm_workgroup_head_id = 0;
93 workgroup_comm = parent_comm;
94 workgroup_comm_id = 0;
95 workgroup_comm_size = 1;
96 mpi_comm_leaders = parent_comm;
97 workgroup_count = 1;
98 workgroup_id = 0;
99 return;
100 }
101
102 tmp_path = build_tmp_dir_path(app_name);
103 parent_comm_workgroup_head_id = identify_workgroup_head();
104
105 assert(MPI_SUCCESS == MPI_Comm_split(
106 parent_comm,
107 parent_comm_workgroup_head_id,
108 parent_comm_id,
109 &workgroup_comm));
110
111 workgroup_comm_id = mpi_get_comm_rank(workgroup_comm);
112 workgroup_comm_size = mpi_get_comm_size(workgroup_comm);
113
114 const bool is_head = is_workgroup_head();
115 assert(MPI_SUCCESS == MPI_Comm_split(
116 parent_comm,
117 is_head ? 0 : 1,
118 parent_comm_id,
119 &mpi_comm_leaders));
120
121 if (!is_head) {
122 assert(MPI_SUCCESS == MPI_Comm_disconnect(&mpi_comm_leaders));
123 mpi_comm_leaders = MPI_COMM_NULL;
124 }
125 assert(MPI_SUCCESS == MPI_Barrier(parent_comm));
126
127 int bcast[2];
128 if (is_head) {
129 bcast[0] = mpi_get_comm_size(mpi_comm_leaders);
130 bcast[1] = mpi_get_comm_rank(mpi_comm_leaders);
131 }
132
133 assert(MPI_SUCCESS == MPI_Bcast(
134 &bcast, 2, get_mpi<int>::type(), 0, workgroup_comm));
135
136 workgroup_count = bcast[0];
137 workgroup_id = bcast[1];
138
139 if (0 == parent_comm_id) { assert(is_head); }
140}

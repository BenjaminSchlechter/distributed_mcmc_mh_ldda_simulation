207 auto execute_rpc(const rpc_message_t &msg)
208 {
209 // std::cout << "execute_rpc: " <<
210 // msg.function_name << std::endl;
211
212 assert(!msg.function_name.empty());
213
214 auto fsearch = function_map.find(msg.function_name);
215 if (function_map.end() == fsearch) {
216 std::cerr << "Unknown remote procedure call: "
217 << msg.function_name << std::endl;
218 assert(false && "Unknown remote procedure call!");
219 } else {
220 // std::cout << "rpc executing: "
221 // << msg.function_name << std::endl;
222 }
223
224 rpc_message_t result_msg;
225 result_msg.id = msg.id;
226 result_msg.function_name =
227 msg.function_name + FCT_NAME_RESULT_TAG;
228 result_msg.answer_required = false;
229
230 abstract_function_t *fptr = (fsearch->second).get();
231
232 try {
233 result_msg.data =
234 fptr->execute(type_abstraction, msg.data);
235 } catch (...) {
236 std::cout << "exception executing function: "
237 << result_msg.function_name << std::endl;

239 }

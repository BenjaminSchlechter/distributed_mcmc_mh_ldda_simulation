20static prec_t bitvector_to_num(const cfg_vector_t &v)
21{
22 prec_t r = 0.0;
23 using size_type = cfg_vector_t::size_type;
24 for (size_type i = 0; i < v.data.size(); i++) {
25 r *= 2.0;
26 r += v[i];
27 }

29 return r / powl(2.0, v.data.size());

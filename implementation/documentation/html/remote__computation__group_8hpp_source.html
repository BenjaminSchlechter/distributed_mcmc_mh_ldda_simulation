203 {
204 MPI_Status status;
205 while (is_message_available(env_comm, &status, mpi_tag)) {
206 if (0 == get_message_size(&status)) {

214 background_comm.test_for_messages();
215
216 if (wait_for_work && (1 < env_comm_size)) {
217 if (!background_comm.communication_is_done()) {

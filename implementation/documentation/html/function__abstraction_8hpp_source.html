259class ConcreteFunction : public AbstractFunction<type_abstraction_adapter_t>
260{
261 private:
263 function_t function;
264
266 param_t parameters;
267
269 result_t result;
270
271 public:
273 ConcreteFunction(auto function, auto param, auto result) :
274 function(function), parameters(param), result(result) {}
275
277 virtual type_abstraction_adapter_t::abstract_type execute(
278 type_abstraction_adapter_t &type_abstraction,
279 const type_abstraction_adapter_t::abstract_type &abstract_param
280 ) override
281 {
282 assert(parameters.deserialize(type_abstraction, abstract_param)
283 && "(most probably wrong parameters passed)");
284
285 using ret_t = decltype(std::apply(function, parameters.values));
286
287 if constexpr (std::is_same<void, ret_t>::value) {
288 std::apply(function, parameters.values);
289 } else {
290 result.values = std::make_tuple(
291 std::apply(function, parameters.values));
292 }
293
294 return result.serialize(type_abstraction);
295 }
296
298 virtual ~ConcreteFunction() { }
299};

259 template<typename scheduler_t,
260 typename result_projection_t = std::identity,
261 typename result_function_t = std::function<void(prec_t)>>
262 void start_computation(
263 prng_t &main_prng,
264 const cfg_vector_t &v,
265 scheduler_t &scheduler,
266 value_comp_t &value_comp,
267 result_projection_t &result_projection = {})
268 {
269 assert(!computation_scheduled_state);
270 computation_scheduled_state = true;
271
272 ignore(&scheduler);
273
274
275 // std::cout << "main_prng: " << main_prng << std::endl;
276 state_data[new_index].prng = derive_prng(main_prng);
277 // std::cout << "derived prng: "
278 // << state_data[new_index].prng() << ", "
279 // << state_data[new_index].prng() << ", "
280 // << state_data[new_index].prng() << ", "
281 // << state_data[new_index].prng << std::endl;
282
283 prob_distribution.reset();
284 z = nextz;
285 nextz = prob_distribution(state_data[new_index].prng);
286 // std::cout << "start_computation: z: "
287 // << z << ", v: " << v << std::endl;
288
289 value_comp.prepare_computation_with_cfg_vector(
290 v, state_data[new_index].pcom);
291
292 state_data[new_index].iteration = 0;
293 state_data[new_index].statistics =
294 basic_statistical_metrics<prec_t>();
295
296 // reset statistics but not already computed values
297 uint8_t old_index = 1 - new_index;
298 state_data[old_index].statistics =
299 basic_statistical_metrics<prec_t>();
300 accept = false;
301
302 schedule_computation =
303 [&](void *scheduler_ptr, std::size_t i, uint8_t index)
304 {
305 scheduler_t &scheduler =
306 *(static_cast<scheduler_t *>(scheduler_ptr));
307
308 result_function_t result_fct
309 = [&, i, index](auto... results)
310 {
311 remaining_computations--;
312 auto result = result_projection(results...);
313 static_assert(
314 std::is_same<prec_t, decltype(result)>::value
315 && "result_projection has wrong result type!");
316 // std::cout << "c R state_data[" << (int) index
317 // << "].values[" << i << "] = "
318 // << result << std::endl;
319 state_data[index].values[i] = result;
320 };
321
322 auto cmd = value_comp(
323 state_data[index].prng,
324 scheduler.rec.rpmanager,
325 state_data[index].pcom);
326 cmd = scheduler.rec.on_result(cmd, result_fct);
327
328 remaining_computations++;
329 scheduler.schedule(cmd);
330 };

341 template<typename scheduler_t>
342 bool continue_computation(
343 scheduler_t &scheduler,
344 value_comp_t &value_comp,
345 acc_prob_comp_t &acc_prob_fct)
346 {
347 assert(computation_scheduled_state);
348 ignore(&value_comp);
349
350 if (is_first_cfg_vector) {
351 return true; // nothing to do
352 }
353
354 if (0 < remaining_computations) {
355 return false; // not yet done
356 }
357
358 uint8_t old_index = 1 - new_index;
359 uint8_t indices[2] = {new_index, old_index};
360
361 // update statistics
362 std::size_t num_values =
363 state_data[new_index].iteration * values_per_iteration;
364 if (0 < num_values) {
365 for (auto j = 0; j < 2; j++) {
366 uint8_t index = indices[j];
367
368 auto it_start =
369 state_data[index].values.begin()
370 + state_data[index].statistics.counted();
371 auto it_end =
372 state_data[index].values.begin() + num_values;
373 state_data[index].statistics(it_start, it_end);
374
375 #if DEBUG_ASSERTIONS
376 assert(num_values
377 == state_data[index].statistics.counted());
378 #endif
379 }
380
381 // test for evidence of result
382 std::size_t num_values_available =
383 std::min(state_data[new_index].statistics.n,
384 state_data[old_index].statistics.n);
385
386 if (num_values_available >= values_required) {
387 auto [p_avg, p_min, p_max] = compute_p_range(
388 coincidence_interval_factor,
389 acc_prob_fct,
390 state_data[new_index].statistics,
391 state_data[old_index].statistics);
392
393 // std::cout << "\t" << "p: [" << p_min << ", "
394 // << p_avg << ", " << p_max << "]" << std::endl;
395
396 // test for statistical evidence of accept/decline
397 if (z <= p_min) {
398 // std::cout << "clear accept ("
399 // << z << ", " << p_min << ")" << std::endl;
400 accept = true;
401 return true;
402 }
403
404 if (z > p_max) {
405 // std::cout << "clear decline ("
406 // << z << ", " << p_max << ")" << std::endl;
407 accept = false;
408 return true;
409 }
410
411 const bool max_iterations_reached =
412 max_iterations <= state_data[new_index].iteration;
413 if (max_iterations_reached) {
414 // std::cout << "max iterations reached ("
415 // << z << ", " << p_avg << ")" << std::endl;
416 accept = (z <= p_avg); // false
417 return true; // done with best guess
418 }
419
420 const bool testniveau_unterschritten =
421 std::abs(p_max - p_min) < testniveau;
422 if (testniveau_unterschritten) {
423 // std::cout << "testniveau_unterschritten ("
424 // << z << ", " << p_avg << ")" << std::endl;
425 accept = (z <= p_avg); // false
426 return true; // done with best guess
427 }
428 }
429 }
430
431 // not yet done => schedule next iteration(s) for more data
432 const bool new_values_for_old_index_are_required =
433 state_data[new_index].iteration
434 >= state_data[old_index].iteration;
435
436 assert(state_data[new_index].iteration
437 <= state_data[old_index].iteration);
438
439 for (auto j = 0;
440 j < (new_values_for_old_index_are_required ? 2 : 1); j++)
441 {
442 uint8_t index = indices[j];
443 state_data[index].iteration++;
444
445 std::size_t num_values =
446 state_data[index].iteration*values_per_iteration;
447 if (num_values > state_data[index].values.size()) {
448 state_data[index].values.resize(num_values);
449 }
450
451 for (std::size_t i = num_values - values_per_iteration;
452 i < num_values; i++)
453 {
454 schedule_computation(
455 static_cast<void *>(&scheduler), i, index);
456 }
457 }
458
459 return false;
460 }
